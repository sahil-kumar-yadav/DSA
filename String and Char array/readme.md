# Reverse a String

```cpp
void reverse(string &str)
{
    int i = 0, j = str.length() - 1;
    cout << "orginal " << str << endl;
    while (i < j)
    {
        swap(str[i], str[j]);
        i++;
        j--;
    }
    cout << "reversed " << str << endl;
}

```

# Check Palindrome

```cpp
bool checkPalindrome(string str)
{

    int i = 0, j = str.length()-1;
    while (i < j)
    {
        if (str[i] != str[j])
        {
            return false;
        }
        i++, j--;
    }
    return true;
}
```
